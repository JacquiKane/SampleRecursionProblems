Binary String To Decimal
binary string -> decimal 

public int binStringToDec(String binStr) {
    // The base case
    if (binStr.length() == 0) 
        return 0;
    // Recursive case
    int first = Integer.parseInt(binStr.substring(0, 1));
    String restOfBinNum = binStr.substring(1);
    return ((int) Math.pow(2 * first, binStr.length() - 1)
            + binStringToDec(restOfBinNum));
}

Decimal to Binary String
decimal to binary string 

public String decToBinString(int val) {
    // Base case
    if (val <= 0) 
        return "";
    else { // Recursive case
        int digit = val % 2;
        return decToBinString(val / 2) + 
                "" + digit;
    }

}

Palindrome?
is string a palindrome? 
public boolean isPalindrome(String str)
{   
  if(str.length() == 0 || str.length() == 1) // Base case
    return true; 
    // Recursive case
  String first = str.substring(0,1);
  String last = str.substring(str.length()-1, str.length());
  if(first.equals(last))
   return isPalindrome(str.substring(1, str.length()-1));
  return false;
}
Pesky Consecutive Duplicates!
get rid of consecutive duplicates ... 

public String dupsRemover(String str) {
 // Base case
 if (str.length() <= 1) 
    return str;
 // Recursive case
 if (str.substring(0, 1).equals(str.substring(1, 2))) 
    return dupsRemover(str.substring(1));    
 else 
    return str.substring(0, 1)
     + dupsRemover(str.substring(1));
}

public int reverseNumber(int n, int totalDigits) {
 if (n <= 9 ) // Base case
    return n;
 else { // Recursive case         
    int lastDigit = n % 10;
    int firstDigits = n/10;
    return (int)((lastDigit*Math.pow(10, totalDigits-1) +
       reverseNumber(firstDigits, totalDigits-1)));
}
}

public String stringReverser(String str) {
    if (str.length() <= 1) // Base case
        return str;

    // Recursive case
    return stringReverser(str.substring(1)) +
